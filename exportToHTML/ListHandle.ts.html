import Editor from '../../../editor'
import { DomElement } from '../../../utils/dom-core'
import SelectionRangeElem from '../SelectionRangeElem'

export type HandlerListOptions = {

    listTarget: string
    $selectionElem: DomElement

    $endElem: DomElement

}
